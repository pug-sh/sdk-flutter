# Typed Track API Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Deliver a compile-time typed event-tracking surface (`Pug.track.purchase(...)`, etc.) under the existing `Pug.track` member, codegen'd from the buf catalog, while keeping `Pug.track('kind', props: {...})` as a discouraged-but-functional escape hatch.

**Architecture:** A callable `TrackNamespace` class (Dart `call` method) exposes `Pug.track(...)` AND `Pug.track.<event>(...)` from a single static field on `Pug`. The class is codegen'd into `lib/src/track_namespace.dart` from `*.pb.dart` descriptors and committed to the repo. All paths land in the existing `PugClient.track(kind, props:)` — runtime, queue, transport, sessions are untouched.

**Tech Stack:** Dart `>=3.7.0`, Flutter `>=3.29.0`, protobuf runtime, `buf` CLI (new tooling), existing `protoc-gen-dart`, `flutter_test`.

**Spec reference:** `docs/superpowers/specs/2026-05-19-typed-track-api-design.md`.

**Branch:** `feat/typed-track-api` (already created).

// AUTOGENERATED — DO NOT EDIT.
// Hand-scaffolded during Task 5; replaced by codegen in Task 7.

import 'configuration.dart';
import 'pug.dart';

class TrackNamespace {
  const TrackNamespace(this._pug);
  final Pug _pug;

  void call(
    String kind, {
    Map<String, Object?> props = const {},
    TrackOptions options = const TrackOptions(),
  }) {
    _pug.capture(kind, props: props, options: options);
  }

  void purchase({
    required String productId,
    required double amount,
    required String currency,
    Map<String, Object?> extras = const {},
    TrackOptions options = const TrackOptions(),
  }) {
    _pug.capture(
      'purchase',
      props: _merge(
        eventName: 'purchase',
        explicit: <String, Object?>{
          'productId': productId,
          'amount': amount,
          'currency': currency,
        },
        extras: extras,
      ),
      options: options,
    );
  }

  Map<String, Object?> _merge({
    required String eventName,
    required Map<String, Object?> explicit,
    required Map<String, Object?> extras,
  }) {
    if (extras.isEmpty) return explicit;
    final result = <String, Object?>{};
    for (final entry in extras.entries) {
      if (explicit.containsKey(entry.key)) {
        _pug.logger.debug(
          'Pug.track.$eventName: extras key "${entry.key}" overridden by explicit named arg.',
        );
        continue;
      }
      result[entry.key] = entry.value;
    }
    result.addAll(explicit);
    return result;
  }
}
```

  void call(
    String kind, {
    Map<String, Object?> props = const {},
    TrackOptions options = const TrackOptions(),
  }) {
    _warnDiscouragedPathOnce(kind);
    _pug.capture(kind, props: props, options: options);
  }

  static final Set<String> _hintedKinds = <String>{};

  void _warnDiscouragedPathOnce(String kind) {
    if (!wellKnownEventSchemas.containsKey(kind)) return;
    if (!_hintedKinds.add(kind)) return;
    final dartName = _kindToDartMethodName(kind);
    _pug.logger.debug(
      "Pug.track('$kind', ...) — consider Pug.track.$dartName(...) for "
      'compile-time typed properties.',
    );
  }

  static String _kindToDartMethodName(String snake) {
    final parts = snake.split('_');
    if (parts.length == 1) return parts.first;
    return parts.first +
        parts.skip(1).map((p) {
          if (p.isEmpty) return '';
          return p[0].toUpperCase() + p.substring(1);
        }).join();
  }
```

//
// Invoke: dart run tool/generate_track_namespace.dart
//
// Reads: lib/src/gen/common/events/**/*.pb.dart (and any other generated
// Properties-bearing locations).
// Writes: lib/src/track_namespace.dart and lib/src/well_known_events.dart.

import 'dart:io';

/// Override table for cases where the convention <X>Properties → snake_case(X)
/// does not produce the desired event kind. Keys are class names; values are
/// the canonical event kind strings.
const Map<String, String> _kindOverrides = <String, String>{
  // Populate as needed during execution; empty by default.
};

/// Override table for Dart-keyword field names. Keys are proto field names;
/// values are the Dart-safe substitutes.
const Map<String, String> _fieldNameOverrides = <String, String>{
  'class': 'cssClass', // 'class' is a Dart keyword
};

void main() async {
  final genRoot = Directory('lib/src/gen/common');
  if (!genRoot.existsSync()) {
    stderr.writeln('Missing $genRoot. Run `make protos` first.');
    exit(1);
  }

  final propertiesClasses = <_PropertiesClass>[];
  for (final file in genRoot.listSync(recursive: true).whereType<File>()) {
    if (!file.path.endsWith('.pb.dart')) continue;
    propertiesClasses.addAll(await _parsePropertiesClasses(file));
  }
  propertiesClasses.sort((a, b) => a.eventKind.compareTo(b.eventKind));

  if (propertiesClasses.isEmpty) {
    stderr.writeln('No Properties classes found under $genRoot.');
    exit(1);
  }

  stderr.writeln('Found ${propertiesClasses.length} Properties classes.');

  await File('lib/src/track_namespace.dart')
      .writeAsString(_renderTrackNamespace(propertiesClasses));
  await File('lib/src/well_known_events.dart')
      .writeAsString(_renderWellKnownEvents(propertiesClasses));
  await _patchEventSchemas('lib/src/events.dart', propertiesClasses);

  stderr.writeln('Wrote lib/src/track_namespace.dart');
  stderr.writeln('Wrote lib/src/well_known_events.dart');
  stderr.writeln('Patched lib/src/events.dart wellKnownEventSchemas block');
}

class _PropertiesClass {
  _PropertiesClass({
    required this.className,
    required this.eventKind,
    required this.fields,
  });

  final String className;
  final String eventKind;
  final List<_PropertyField> fields;
}

class _PropertyField {
  _PropertyField({
    required this.protoName,
    required this.dartName,
    required this.dartType,
    required this.kind,
    this.required = false,
  });

  final String protoName;
  final String dartName;
  final String dartType; // 'String' | 'int' | 'double' | 'bool'
  bool required; // set after construction by _requiredFieldsForClass
  final String kind; // 'string' | 'int32' | 'double' | 'bool'
}

Future<List<_PropertiesClass>> _parsePropertiesClasses(File file) async {
  // Parses `.pb.dart` to extract Properties classes and their fields.
  //
  // Strategy: read the file as text, regex-find `class XxxProperties extends
  // $pb.GeneratedMessage`, then locate the FieldInfo registrations inside
  // the class body. For each field, extract:
  //   - the proto field name (from `protoName: '...'`)
  //   - the scalar type (from `..a<...>(..., $pb.PbFieldType.O3, ...)` or
  //     the `aOS`/`a(...)` helper)
  //   - whether it's required (presence of `..pPS` style or check the proto
  //     option file separately — see below)
  //
  // The `required` flag is *not* directly readable from .pb.dart;
  // protoc-gen-dart does not emit `(buf.validate.field).required` annotations.
  // We have two options:
  //   (a) Parse the source .proto files for the `required = true` option.
  //   (b) Maintain a manual required-field table (small; one per Properties class).
  //
  // For an MVP, this scaffold uses approach (a). The implementer
  // should fill in the regex-driven extraction here.
  final source = await file.readAsString();
  final classMatches = RegExp(
    r'class (\w+Properties) extends \$pb\.GeneratedMessage',
  ).allMatches(source);

  final results = <_PropertiesClass>[];
  for (final match in classMatches) {
    final className = match.group(1)!;
    final eventKind =
        _kindOverrides[className] ?? _classNameToKind(className);

    // Extract the class body — naive: from class match to the next `}\n\n`.
    final classStart = match.start;
    final classEnd = source.indexOf('\n}\n', classStart);
    if (classEnd == -1) continue;
    final classBody = source.substring(classStart, classEnd);

    final fields = _extractFields(classBody);

    // Look up the matching .proto file for required-field info.
    final requiredFields = await _requiredFieldsForClass(className);
    for (final f in fields) {
      f.required = requiredFields.contains(f.protoName);
    }

    results.add(_PropertiesClass(
      className: className,
      eventKind: eventKind,
      fields: fields,
    ));
  }
  return results;
}

String _classNameToKind(String camelClassName) {
  final stripped = camelClassName.endsWith('Properties')
      ? camelClassName.substring(0, camelClassName.length - 'Properties'.length)
      : camelClassName;
  final buf = StringBuffer();
  for (var i = 0; i < stripped.length; i++) {
    final c = stripped[i];
    final isUpper = c == c.toUpperCase() && c != c.toLowerCase();
    if (isUpper && i > 0) buf.write('_');
    buf.write(c.toLowerCase());
  }
  return buf.toString();
}

List<_PropertyField> _extractFields(String classBody) {
  // Each scalar field appears as `..a<core.String>(1, 'class', $pb.PbFieldType.OS)`
  // or similar. Capture the protoName and type.
  final fieldRe = RegExp(
    r"\.\.a<(core\.)?(\w+)>\(\d+, '(\w+)', \$pb\.PbFieldType\.(\w+)\)",
  );
  final fields = <_PropertyField>[];
  for (final m in fieldRe.allMatches(classBody)) {
    final dartType = m.group(2)!;
    final protoName = m.group(3)!;
    final pbType = m.group(4)!;
    fields.add(_PropertyField(
      protoName: protoName,
      dartName: _fieldNameOverrides[protoName] ?? _snakeToCamel(protoName),
      dartType: _mapDartType(dartType, pbType),
      kind: _pbTypeToKind(pbType),
    ));
  }
  return fields;
}

String _snakeToCamel(String snake) {
  final parts = snake.split('_');
  if (parts.length == 1) return parts.first;
  return parts.first +
      parts.skip(1).map((p) {
        if (p.isEmpty) return '';
        return p[0].toUpperCase() + p.substring(1);
      }).join();
}

String _mapDartType(String dartType, String pbType) {
  // pbType is one of 'OS' (string), 'O3' (int32), 'OD' (double), 'OB' (bool), etc.
  switch (pbType) {
    case 'OS': return 'String';
    case 'O3': return 'int';
    case 'OD': return 'double';
    case 'OB': return 'bool';
    default: return dartType; // fall back to whatever protoc-gen-dart used
  }
}

String _pbTypeToKind(String pbType) {
  switch (pbType) {
    case 'OS': return 'string';
    case 'O3': return 'int32';
    case 'OD': return 'double';
    case 'OB': return 'bool';
    default: return 'string';
  }
}

Future<Set<String>> _requiredFieldsForClass(String className) async {
  // Parse the corresponding .proto file. The proto root is `proto/`.
  // Naive approach: search all .proto files for `message <ClassName>` and
  // then within that block, collect fields with `(buf.validate.field).required = true`.
  for (final file in Directory('proto')
      .listSync(recursive: true)
      .whereType<File>()
      .where((f) => f.path.endsWith('.proto'))) {
    final source = await file.readAsString();
    final messageRe = RegExp(
      r'message ' + className + r'\s*\{([^}]*)\}',
      multiLine: true,
      dotAll: true,
    );
    final match = messageRe.firstMatch(source);
    if (match == null) continue;
    final body = match.group(1)!;
    final requiredFields = <String>{};
    final fieldRe = RegExp(
      r'(\w+)\s+(\w+)\s*=\s*\d+\s*\[[^\]]*\(buf\.validate\.field\)\.required\s*=\s*true',
      multiLine: true,
      dotAll: true,
    );
    for (final m in fieldRe.allMatches(body)) {
      requiredFields.add(m.group(2)!);
    }
    return requiredFields;
  }
  return <String>{};
}

  buf.writeln('}');

Future<void> _patchEventSchemas(
  String path,
  List<_PropertiesClass> classes,
) async {
  final file = File(path);
  final source = await file.readAsString();
  final startMarker =
      'const wellKnownEventSchemas = <String, WellKnownEventSchema>{';
  final endMarker = '};';
  final startIdx = source.indexOf(startMarker);
  if (startIdx == -1) {
    stderr.writeln('Could not find wellKnownEventSchemas in $path');
    exit(1);
  }
  final endIdx = source.indexOf(endMarker, startIdx);
  if (endIdx == -1) {
    stderr.writeln('Could not find closing brace for wellKnownEventSchemas in $path');
    exit(1);
  }

  final buf = StringBuffer();
  buf.writeln(startMarker);
  for (final c in classes) {
    buf.writeln("  '${c.eventKind}': WellKnownEventSchema({");
    for (final f in c.fields) {
      final fieldVar = switch (f.kind) {
        'string' => '_stringField',
        'int32' => '_intField',
        'double' => '_doubleField',
        'bool' => '_boolField',
        _ => '_stringField',
      };
      buf.writeln("    '${f.dartName}': $fieldVar,");
    }
    buf.writeln('  }),');
  }
  buf.write('}');

  final patched =
      source.substring(0, startIdx) + buf.toString() + source.substring(endIdx + 1);
  await file.writeAsString(patched);
}
